# Save this file as jira_data_utility.pyw import tkinter as tk from tkinter import ttk, messagebox from tkcalendar import DateEntry import datetime import os import requests import pandas as pd import xlsxwriter import numpy as np import threading import ttkbootstrap as ttkb from ttkbootstrap.constants import * # Set the working directory to the script's directory script_dir = os.path.dirname(os.path.abspath(__file__)) os.chdir(script_dir) class JiraDataUtility: def __init__(self, root): self.root = root self.root.title("Jira Data Utility") self.root.geometry("600x500") self.root.resizable(True, True) # Use ttkbootstrap style style = ttkb.Style(theme="flatly") # Modern theme # Main frame self.main_frame = ttk.Frame(self.root, padding=10) self.main_frame.pack(fill=tk.BOTH, expand=True) # Date inputs frame self.date_frame = ttk.Frame(self.main_frame) self.date_frame.pack(pady=5, fill=tk.X) # Default dates today = datetime.date.today() yesterday = today - datetime.timedelta(days=1) if today.weekday() == 0: # If Monday, set end date to previous Friday default_end = today - datetime.timedelta(days=3) else: default_end = yesterday # Start date picker ttk.Label(self.date_frame, text="Start Date:").pack(side=tk.LEFT, padx=5) self.start_date = DateEntry( self.date_frame, width=12, date_pattern='yyyy-mm-dd', year=yesterday.year, month=yesterday.month, day=yesterday.day ) self.start_date.pack(side=tk.LEFT, padx=5) # End date picker ttk.Label(self.date_frame, text="End Date:").pack(side=tk.LEFT, padx=5) self.end_date = DateEntry( self.date_frame, width=12, date_pattern='yyyy-mm-dd', year=default_end.year, month=default_end.month, day=default_end.day ) self.end_date.pack(side=tk.LEFT, padx=5) # Controls frame self.controls_frame = ttk.Frame(self.main_frame) self.controls_frame.pack(pady=10, fill=tk.X) # Start button self.start_button = ttk.Button( self.controls_frame, text="Start", command=self.start_process, bootstyle=SUCCESS ) self.start_button.pack(side=tk.LEFT, padx=5) # Loading label self.loading_label = ttk.Label( self.controls_frame, text="", bootstyle=SECONDARY ) self.loading_label.pack(side=tk.LEFT, padx=10) # Progress bar self.progress = ttk.Progressbar( self.controls_frame, length=300, mode='determinate', bootstyle=INFO ) self.progress.pack(side=tk.LEFT, padx=5) # Log area self.log_frame = ttk.Frame(self.main_frame) self.log_frame.pack(fill=tk.BOTH, expand=True, padx=5, pady=5) self.log_text = tk.Text( self.log_frame, height=15, width=70, state='disabled', wrap=tk.WORD, font=("Segoe UI", 10) ) self.log_text.pack(side=tk.LEFT, fill=tk.BOTH, expand=True) # Scrollbar for log text scrollbar = ttk.Scrollbar( self.log_frame, orient=tk.VERTICAL, command=self.log_text.yview, bootstyle=SECONDARY ) scrollbar.pack(side=tk.RIGHT, fill=tk.Y) self.log_text['yscrollcommand'] = scrollbar.set # API Tokens self.crm_token = "crm_token" # Replace with actual token self.jira_token = "jira_token" # Replace with actual token self.crm_url = "https://demo.defectdojo.org/api/v2/users/?is_active=true&limit=100" self.jira_server = "https://jira.crm.com" # Initialize JQL templates and display names self.jql_templates = {} self.jql_display_names = {} def log(self, message): """Log messages to GUI text area.""" self.log_text.config(state='normal') timestamp = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S') self.log_text.insert(tk.END, f"{timestamp}: {message}\n") self.log_text.config(state='disabled') self.log_text.see(tk.END) self.root.update() def start_process(self): """Main process triggered by Start button.""" self.start_button.config(state='disabled') self.loading_label.config(text="Loading") self.progress['value'] = 0 self.animate_loading() threading.Thread(target=self.process_thread, daemon=True).start() def animate_loading(self): """Animate loading dots in the loading label.""" def update_dots(): if self.loading_label.cget("text").startswith("Loading"): dots = self.loading_label.cget("text")[7:] if dots == "...": self.loading_label.config(text="Loading") else: self.loading_label.config(text="Loading" + dots + ".") self.root.after(1000, update_dots) self.root.after(1000, update_dots) def process_thread(self): """Run the main process in a separate thread.""" try: start_date = self.start_date.get_date().strftime('%Y-%m-%d') end_date = self.end_date.get_date().strftime('%Y-%m-%d') self.log("Process started.") # Step 1: Download usernames (25%) self.progress['value'] = 10 usernames = self.download_usernames() if not usernames: raise ValueError("No usernames retrieved.") self.progress['value'] = 25 # Step 2: Fetch JQL data (50%) self.log("Fetching JQL data...") jql_files = self.fetch_jql_data(start_date, end_date, usernames) self.progress['value'] = 50 # Step 3: Merge excels (75%) self.log("Merging Excel files...") self.merge_excels(jql_files, start_date, end_date) self.progress['value'] = 75 # Step 4: Update daily finals.xlsx (100%) self.log("Updating finals.xlsx...") self.update_daily_final(start_date, end_date) self.progress['value'] = 100 self.log("Process completed successfully.") self.root.after(1, lambda: messagebox.showinfo("info", "Process completed successfully")) except Exception as e: self.log(f"Error: {str(e)}") self.root.after( 1, lambda err=e: messagebox.showerror("Error", f"An error occurred: {str(err)}") ) finally: self.loading_label.config(text="") self.start_button.config(state="normal") def download_usernames(self): """Step 1: Download usernames from DefectDojo CRM API and save to Excel.""" self.log("Downloading users from DefectDojo CRM API...") headers = {"Authorization": f"Token {self.crm_token}"} usernames = [] url = self.crm_url page = 1 max_pages = 10 try: while url and page <= max_pages: self.log(f"Fetching page {page} from {url}"...") response = requests.get(url, headers=headers, timeout=10) response.raise_for_status() data = response.json() self.log(f"Extracted {len(data.get('results', []))} users from page {page}...") if 'results' not in data: raise ValueError(f"JSON response does not contain 'results' key. Response: {data}") for user in data.get('results', []): email = user.get('email', '') if email and '@' in email: username = email.split('@')[0] usernames.append(username) else: self.log(f"Skipping user with invalid email: {user}") url = data.get('next', None) page += 1 if not usernames: raise ValueError("No valid usernames found in API response") output_file = "usernames.xlsx" if os.path.exists(output_file): os.remove(output_file) self.log(f"Deleted existing {output_file}") df = pd.DataFrame(usernames, columns=["Username"]) df.to_excel(output_file, index=False, engine='openpyxl') self.log(f"Saved {len(usernames)} usernames to {output_file}") return usernames except requests.exceptions.HTTPError as http_err: self.log(f"HTTP error occurred: {http_err}, Response: {response.text}") raise except requests.exceptions.RequestException as req_err: self.log(f"Request error occurred: {req_err}") raise except ValueError as val_err: self.log(f"Value error: {val_err}") raise except Exception as e: self.log(f"Unexpected error downloading usernames: {str(e)}") raise def fetch_jql_data(self, start_date, end_date, usernames): """Step 2: Fetch JQL data using HTTP requests with Bearer token and save to jqlX.xlsx.""" self.log("Fetching JQL data...") jql_files = [] # Define JQL queries and display names self.jql_templates = { "jql1": 'status changed FROM "FS-SECURITY REVIEW PENDING" TO "FS-QA REVIEW PENDING" DURING ("{start_date}", "{end_date}") BY {username}', "jql2": 'status changed FROM "FS-SECURITY REVIEW PENDING" TO "FUNCTIONAL SPECIFICATION PENDING" DURING ("{start_date}", "{end_date}") BY {username}', "jql3": 'status changed FROM "DS-SECURITY REVIEW PENDING" TO "DESIGN SPECIFICATION ACCEPTED" DURING ("{start_date}", "{end_date}") BY {username}', "jql4": 'status changed FROM "TO DO" TO "SS-SECURITY REVIEW PENDING" to "DO" DURING ("{start_date}", "{end_date}") BY {username}', "jql5": 'status changed FROM "READY FOR SECURITY" TO "READY FOR PAINT" DURING ("{start_date}", "{end_date}") BY {start_date}', "jql6": 'status changed FROM "REOPENED" TO "REOPENED" DURING ("start_date}", "{start_date}") BY {end_date}', } self.jql_display_names = { "jql1": "FS Accepted", "jql2": "FS Rejected", "jql3": "DS Accepted", "jql4": "DS Rejected", "jql5": "RFS Accepted", "jql6": "RFS Rejected", } headers = { "Authorization": f"Bearer {self.jira_token}", "Content-Type": "application/json" } jira_base_url = f"{self.jira_server}/rest/api/2/search" for jql_name, jql_template in self.jql_templates.items(): issues_list = [] for username in usernames[:100]: jql = jql_template.format(start_date=start_date, end_date=end_date, username=username) self.log(f"Executing {jql_name} ({self.jql_display_names.get(jql_name, jql_name)}) for {username}") try: start_at = 0 max_results = start_at while True: payload = {"jql": jql, "startAt": start_at, "maxResults": max_results} response = requests.post(jira_base_url, headers=headers, json=payload, timeout=10) response.raise_for_status() data = response.json() if "issues" not in data: self.log(f"No 'issues' key in response for {jql_name}: {data}") break for issue in data.get("issues", []): issues_list.append({ "Source": jql_name, "Issue Key": issue.get("key", ""), "Status": issue.get("fields", {}).get("status", {}).get("name", ""), "Project Name": issue.get("fields", {}).get("project", {}).get("name", ""), "Priority": issue.get("fields", {}).get("priority", {}).get("name", "") }) total = data.get("total", 0) start_at += max_results if start_at >= total: break except requests.exceptions.HTTPError as http_err: self.log(f"HTTP error fetching {jql_name} for {username}: {http_err}") continue except Exception as e: self.log(f"Error fetching {jql_name} for {username}: {str(e)}") continue if issues_list: file_path = f"{jql_name}.xlsx" if os.path.exists(file_path): os.remove(file_path) self.log(f"Deleted existing {file_path}") df = pd.DataFrame(issues_list) df.to_excel(file_path, index=False, engine='openpyxl') jql_files.append(file_path) self.log(f"Saved {len(issues_list)} issues to {file_path}") return jql_files def merge_excels(self, jql_files, start_date, end_date): """Step 3: Merge excels into Date_{startdate_enddate}.xlsx.""" self.log("Merging Excel files...") merged_data = pd.DataFrame() for file in jql_files: df = pd.read_excel(file, engine='openpyxl') merged_data = pd.concat([merged_data, df], ignore_index=True) output_file = f"Date_{start_date}_{end_date}.xlsx" if os.path.exists(output_file): os.remove(output_file) self.log(f"Deleted existing {output_file}") workbook = xlsxwriter.Workbook(output_file) worksheet = workbook.add_worksheet() center_format = workbook.add_format({'align': 'center'}) headers = ["Source", "Issue Key", "Status", "Project Name", "Priority"] for col, header in enumerate(headers): worksheet.write(0, col, header, center_format) for row, data in enumerate(merged_data.itertuples(index=False), 1): for col, value in enumerate(data): worksheet.write(row, col, value, center_format) workbook.close() self.log(f"Merged data saved to {output_file}") def update_daily_final(self, start_date, end_date): """Step 4: Update daily finals.xlsx, hide zeros for JQLs, handle NaN/INF, apply bold and borders.""" self.log("Updating finals.xlsx...") date_str = datetime.datetime.strptime(start_date, '%Y-%m-%d').strftime('%d-%b-%Y') jqls = list(self.jql_templates.keys()) # Initialize or load existing finals.xlsx final_file = "finals.xlsx" if os.path.exists(final_file): try: df_final = pd.read_excel(final_file, engine='openpyxl') df_final = df_final.fillna(0) self.log("Loaded existing finals.xlsx and replaced NaN with 0") if np.isinf(df_final.select_dtypes(include=np.number)).any().any(): self.log("Warning: INF values detected in finals.xlsx, replacing with 0") df_final = df_final.replace([np.inf, -np.inf], 0) except Exception as e: self.log(f"Error reading finals.xlsx: {str(e)}") df_final = pd.DataFrame({"Type": [self.jql_display_names.get(jql, jql) for jql in jqls] + ["Total"]}) else: df_final = pd.DataFrame({"Type": [self.jql_display_names.get(jql, jql) for jql in jqls] + ["Total"]}) self.log("Initialized new finals.xlsx") # Count issues for each JQL counts = {} for jql in jqls: file_path = f"{jql}.xlsx" counts[jql] = 0 if os.path.exists(file_path): try: df = pd.read_excel(file_path, engine='openpyxl') counts[jql] = len(df) self.log(f"Counted {counts[jql]} issues in {file_path}") except Exception as e: self.log(f"Error reading {file_path}: {str(e)}") else: self.log(f"File {file_path} does not exist, count set to 0") # Add new date column if not exists if date_str not in df_final.columns: df_final[date_str] = [counts.get(jql, 0) for jql in jqls] + [sum(counts.values())] self.log(f"Added new date column {date_str}") # Save with formatting workbook = xlsxwriter.Workbook(final_file, {'nan_inf_to_errors': True}) worksheet = workbook.add_worksheet() # Define formats bold_center_format = workbook.add_format({ 'bold': True, 'align': 'center', 'border': 1 }) bold_left_format = workbook.add_format({ 'bold': True, 'align': 'left', 'border': 1 }) center_format = workbook.add_format({ 'align': 'center', 'border': 1 }) # Write headers for col, header in enumerate(df_final.columns): format_to_use = bold_left_format if col == 0 else bold_center_format worksheet.write(0, col, header, format_to_use) # Write data, hide zeros for non-Total rows for row, data in enumerate(df_final.itertuples(index=False), 1): is_total_row = data[0] == "Total" for col, value in enumerate(data): try: numeric_value = float(value) if np.isnan(numeric_value) or np.isinf(numeric_value): self.log(f"NaN/INF detected at row {row}, col {col}, replacing with 0") value = 0 except (ValueError, TypeError): pass display_value = value if col > 0 and not is_total_row and value == 0: display_value = "" format_to_use = bold_left_format if col == 0 or is_total_row else center_format worksheet.write(row, col, display_value, format_to_use) workbook.close() self.log(f"Updated {final_file} with data for {date_str}") # Delete intermediate files files_to_delete = [f"{jql}.xlsx" for jql in jqls] + ["usernames.xlsx"] for file in files_to_delete: try: if os.path.exists(file): os.remove(file) self.log(f"Deleted {file}") else: self.log(f"File {file} does not exist, skipping deletion") except Exception as e: self.log(f"Error deleting {file}: {str(e)}") if __name__ == "__main__": root = ttkb.Window() app = JiraDataUtility(root) root.mainloop()
