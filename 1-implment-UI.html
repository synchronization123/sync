import os import pandas as pd # Folder containing xlsx files folder_path = 'your/folder/path' output_file = 'merged.xlsx' all_data = [] for file in os.listdir(folder_path): if file.endswith('.xlsx'): file_path = os.path.join(folder_path, file) df = pd.read_excel(file_path) df['SourceFile'] = file # Optional: add source filename for tracking all_data.append(df) merged_df = pd.concat(all_data, ignore_index=True) merged_df.to_excel(output_file, index=False) print(f"All rows merged into one sheet in {output_file}")
