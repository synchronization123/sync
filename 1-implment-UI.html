import tkinter as tk from tkinter import ttk, messagebox from tkcalendar import DateEntry import datetime import os from pathlib import Path import requests import pandas as pd import xlsxwriter import numpy as np class JiraDataUtility: def __init__(self, root): self.root = root self.root.title("Jira Data Utility") self.root.geometry("700x650") self.root.resizable(True, True) # Set modern theme style = ttk.Style() style.theme_use('clam') # Modern theme: 'clam', 'alt', 'default', or 'classic' # Configure styles style.configure('TLabel', font=('Helvetica', 10)) style.configure('TButton', font=('Helvetica', 10), padding=5) style.configure('TFrame', padding=10) # Main frame self.main_frame = ttk.Frame(self.root) self.main_frame.pack(fill='both', expand=True, padx=10, pady=10) # Default dates today = datetime.date.today() yesterday = today - datetime.timedelta(days=1) if today.weekday() == 0: # If Monday, set end date to previous Friday default_end = today - datetime.timedelta(days=3) else: default_end = yesterday # Datepickers and Start button frame self.input_frame = ttk.Frame(self.main_frame) self.input_frame.pack(fill='x', pady=10) # Start Date ttk.Label(self.input_frame, text="Start Date:").pack(side='left', padx=5) self.start_date = DateEntry( self.input_frame, width=12, date_pattern='yyyy-mm-dd', year=yesterday.year, month=yesterday.month, day=yesterday.day, font=('Helvetica', 10) ) self.start_date.pack(side='left', padx=5) # End Date ttk.Label(self.input_frame, text="End Date:").pack(side='left', padx=5) self.end_date = DateEntry( self.input_frame, width=12, date_pattern='yyyy-mm-dd', year=default_end.year, month=default_end.month, day=default_end.day, font=('Helvetica', 10) ) self.end_date.pack(side='left', padx=5) # Start Button self.start_button = ttk.Button(self.input_frame, text="Start", command=self.start_process) self.start_button.pack(side='left', padx=5) # Console area ttk.Label(self.main_frame, text="Execution Log:").pack(pady=5) self.console_frame = ttk.Frame(self.main_frame) self.console_frame.pack(fill='both', expand=True, pady=5) self.console = tk.Text( self.console_frame, height=20, width=70, wrap='word', font=('Helvetica', 9), borderwidth=1, relief='solid' ) self.console.pack(side='left', fill='both', expand=True) self.console.config(state='disabled') scrollbar = ttk.Scrollbar(self.console_frame, orient='vertical', command=self.console.yview) scrollbar.pack(side='right', fill='y') self.console.config(yscrollcommand=scrollbar.set) # Status bar self.status_var = tk.StringVar(value="Ready") status_bar = ttk.Label( self.main_frame, textvariable=self.status_var, relief='sunken', anchor='w', padding=5 ) status_bar.pack(side='bottom', fill='x') # API Tokens self.crm_token = "crm_token" self.jira_token = "jira_token+" self.crm_url = "https://demo.defectdojo.org/api/v2/users/?is_active=true&limit=100" self.jira_server = "https://jira.crm.com" # Initialize JQL templates and display names self.jql_templates = {} self.jql_display_names = {} # Current working directory and data folder self.working_dir = Path.cwd() self.data_dir = self.working_dir / "data" self.data_dir.mkdir(exist_ok=True) # Create /data folder if it doesn't exist def log(self, message): """Log messages to GUI text area and update status.""" self.console.config(state='normal') timestamp = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S') self.console.insert(tk.END, f"{timestamp}: {message}\n") self.console.see(tk.END) self.console.config(state='disabled') self.status_var.set(message[:50] + "..." if len(message) > 50 else message) self.root.update() def start_process(self): """Main process triggered by Start button.""" self.start_button.config(state='disabled') self.status_var.set("Processing...") try: start_date = self.start_date.get_date().strftime('%Y-%m-%d') end_date = self.end_date.get_date().strftime('%Y-%m-%d') self.log("Process started.") # Step 1: Download usernames usernames = self.download_usernames() if not usernames: raise Exception("No usernames retrieved.") # Step 2: Fetch JQL data jql_files = self.fetch_jql_data(start_date, end_date, usernames) # Step 3: Merge excels self.merge_excels(jql_files, start_date, end_date) # Step 4: Update daily FS-DS-RFS_Report.xlsx self.update_daily_final(start_date, end_date) self.log("Process completed successfully.") self.status_var.set("Process completed successfully") messagebox.showinfo("Success", "Process completed successfully!") except Exception as e: self.log(f"Error: {str(e)}") self.status_var.set("Error occurred") messagebox.showerror("Error", f"An error occurred: {str(e)}") finally: self.start_button.config(state='normal') def download_usernames(self): """Step 1: Download usernames from DefectDojo CRM API and save to usernames.xlsx.""" self.log("Downloading users from DefectDojo CRM API...") headers = {"Authorization": f"Token {self.crm_token}"} usernames = [] url = self.crm_url page = 1 max_pages = 10 try: while url and page <= max_pages: self.log(f"Fetching page {page} from {url}") response = requests.get(url, headers=headers, timeout=10) response.raise_for_status() data = response.json() self.log(f"Extracted {len(data.get('results', []))} users from page {page}") if 'results' not in data: raise ValueError(f"JSON response does not contain 'results' key. Response: {data}") for user in data.get('results', []): email = user.get('email', '') if email and '@' in email: username = email.split('@')[0] usernames.append(username) else: self.log(f"Skipping user with invalid email: {user}") url = data.get('next', None) page += 1 if not usernames: raise ValueError("No valid usernames found in API response") output_file = self.data_dir / "usernames.xlsx" if output_file.exists(): output_file.unlink() self.log(f"Deleted existing {output_file}") df = pd.DataFrame(usernames, columns=["Username"]) df.to_excel(output_file, index=False, engine='openpyxl') self.log(f"Saved {len(usernames)} usernames to {output_file}") return usernames except requests.exceptions.HTTPError as http_err: self.log(f"HTTP error occurred: {http_err}, Response: {response.text}") raise except requests.exceptions.RequestException as req_err: self.log(f"Request error occurred: {req_err}") raise except ValueError as val_err: self.log(f"Value error: {val_err}") raise except Exception as e: self.log(f"Unexpected error downloading usernames: {str(e)}") raise def fetch_jql_data(self, start_date, end_date, usernames): """Step 2: Fetch JQL data using HTTP requests with Bearer token and save to jqlX.xlsx.""" self.log("Fetching JQL data...") jql_files = [] # Define JQL queries and display names self.jql_templates = { "jql1": 'status changed FROM "FS-SECURITY REVIEW PENDING" TO "FS-QA REVIEW PENDING" DURING ("{start_date}", "{end_date}") BY {username}', "jql2": 'status changed FROM "FS-SECURITY REVIEW PENDING" TO "FUNCTIONAL SPECIFICATION PENDING" DURING ("{start_date}", "{end_date}") BY {username}', "jql3": 'status changed FROM "DS-SECURITY REVIEW PENDING" TO "DESIGN SPECIFICATION ACCEPTED" DURING ("{start_date}", "{end_date}") BY {username}', "jql4": 'status changed FROM "DS-SECURITY REVIEW PENDING" TO "TO DO" DURING ("{start_date}", "{end_date}") BY {username}', "jql5": 'status changed FROM "READY FOR SECURITY" TO "READY FOR PA" DURING ("{start_date}", "{end_date}") BY {username}', "jql6": 'status changed FROM "READY FOR SECURITY" TO "REOPENED" DURING ("{start_date}", "{end_date}") BY {username}', } self.jql_display_names = { "jql1": "FS Accepted", "jql2": "FS Rejected", "jql3": "DS Accepted", "jql4": "DS Rejected", "jql5": "RFS Accepted", "jql6": "RFS Rejected", } headers = { "Authorization": f"Bearer {self.jira_token}", "Content-Type": "application/json" } jira_base_url = f"{self.jira_server}/rest/api/2/search" for jql_name, jql_template in self.jql_templates.items(): issues_list = [] for username in usernames[:100]: jql = jql_template.format(start_date=start_date, end_date=end_date, username=username) self.log(f"Executing {jql_name} ({self.jql_display_names.get(jql_name, jql_name)}) for {username}: {jql}") try: start_at = 0 max_results = 50 while True: payload = {"jql": jql, "startAt": start_at, "maxResults": max_results} response = requests.post(jira_base_url, headers=headers, json=payload, timeout=10) response.raise_for_status() data = response.json() if "issues" not in data: self.log(f"No 'issues' key in response for {jql_name}: {data}") break for issue in data.get("issues", []): issues_list.append({ "Source": jql_name, "Issue Key": issue.get("key", ""), "Status": issue.get("fields", {}).get("status", {}).get("name", ""), "Project Name": issue.get("fields", {}).get("project", {}).get("name", ""), "Priority": issue.get("fields", {}).get("priority", {}).get("name", "") }) total = data.get("total", 0) start_at += max_results if start_at >= total: break except requests.exceptions.HTTPError as http_err: self.log(f"HTTP error fetching {jql_name} for {username}: {http_err}, Response: {response.text}") continue except Exception as e: self.log(f"Error fetching {jql_name} for {username}: {str(e)}") continue if issues_list: file_path = self.data_dir / f"{jql_name}.xlsx" if file_path.exists(): file_path.unlink() self.log(f"Deleted existing {file_path}") df = pd.DataFrame(issues_list) df.to_excel(file_path, index=False, engine='openpyxl') jql_files.append(file_path) self.log(f"Saved {len(issues_list)} issues to {file_path}") return jql_files def merge_excels(self, jql_files, start_date, end_date): """Step 3: Merge excels into Date_{startdate_enddate}.xlsx in /data folder.""" self.log("Merging Excel files...") merged_data = pd.DataFrame() for file in jql_files: df = pd.read_excel(file, engine='openpyxl') merged_data = pd.concat([merged_data, df], ignore_index=True) output_file = self.data_dir / f"Date_{start_date}_{end_date}.xlsx" if output_file.exists(): output_file.unlink() self.log(f"Deleted existing {output_file}") workbook = xlsxwriter.Workbook(output_file) worksheet = workbook.add_worksheet() center_format = workbook.add_format({'align': 'center'}) headers = ["Source", "Issue Key", "Status", "Project Name", "Priority"] for col, header in enumerate(headers): worksheet.write(0, col, header, center_format) for row, data in enumerate(merged_data.itertuples(index=False), 1): for col, value in enumerate(data): worksheet.write(row, col, value, center_format) workbook.close() self.log(f"Merged data saved to {output_file}") def update_daily_final(self, start_date, end_date): """Step 4: Update daily FS-DS-RFS_Report.xlsx, hide zeros for JQLs, handle NaN/INF, apply bold and borders, center-align dates, then delete files.""" self.log("Updating daily FS-DS-RFS_Report.xlsx...") date_str = datetime.datetime.strptime(start_date, '%Y-%m-%d').strftime('%d-%b-%Y') jqls = list(self.jql_templates.keys()) # Initialize or load existing FS-DS-RFS_Report.xlsx final_file = self.working_dir / "FS-DS-RFS_Report.xlsx" if final_file.exists(): try: df_final = pd.read_excel(final_file, engine='openpyxl') df_final = df_final.fillna(0) self.log("Loaded existing FS-DS-RFS_Report.xlsx and replaced NaN with 0") if np.isinf(df_final.select_dtypes(include=np.number)).any().any(): self.log("Warning: INF values detected in FS-DS-RFS_Report.xlsx, replacing with 0") df_final = df_final.replace([np.inf, -np.inf], 0) except Exception as e: self.log(f"Error reading FS-DS-RFS_Report.xlsx: {str(e)}, initializing new DataFrame") df_final = pd.DataFrame({"Type": [self.jql_display_names.get(jql, jql) for jql in jqls] + ["Total"]}) else: df_final = pd.DataFrame({"Type": [self.jql_display_names.get(jql, jql) for jql in jqls] + ["Total"]}) self.log("Initialized new FS-DS-RFS_Report.xlsx") # Count issues for each JQL counts = {} for jql in jqls: file_path = self.data_dir / f"{jql}.xlsx" counts[jql] = 0 if file_path.exists(): try: df = pd.read_excel(file_path, engine='openpyxl') counts[jql] = len(df) self.log(f"Counted {counts[jql]} issues in {file_path}") except Exception as e: self.log(f"Error reading {file_path}: {str(e)}") else: self.log(f"File {file_path} does not exist, count set to 0") # Add new date column if not exists if date_str not in df_final.columns: df_final[date_str] = [counts.get(jql, 0) for jql in jqls] + [sum(counts.values())] self.log(f"Added new date column {date_str}") # Save with formatting workbook = xlsxwriter.Workbook(final_file, {'nan_inf_to_errors': True}) worksheet = workbook.add_worksheet() # Define formats bold_center_format = workbook.add_format({ 'bold': True, 'align': 'center', 'border': 1 }) bold_left_format = workbook.add_format({ 'bold': True, 'align': 'left', 'border': 1 }) center_format = workbook.add_format({ 'align': 'center', 'border': 1 }) # Write headers for col, header in enumerate(df_final.columns): format_to_use = bold_left_format if col == 0 else bold_center_format worksheet.write(0, col, header, format_to_use) # Write data, hide zeros for JQLs, center-align all date columns for row, data in enumerate(df_final.itertuples(index=False), 1): for col, value in enumerate(data): try: numeric_value = float(value) if np.isnan(numeric_value) or np.isinf(numeric_value): self.log(f"NaN/INF detected at row {row}, col {col}, replacing with 0") value = 0 except (ValueError, TypeError): pass display_value = value is_total_row = data[0] == "Total" if col > 0 and not is_total_row and value == 0: display_value = "" # Apply center alignment for all date columns, including Total row format_to_use = bold_left_format if col == 0 else center_format worksheet.write(row, col, display_value, format_to_use) workbook.close() self.log(f"Updated {final_file} with data for {date_str}") # Delete intermediate files files_to_delete = [self.data_dir / f"{jql}.xlsx" for jql in jqls] + [self.data_dir / "usernames.xlsx"] for file in files_to_delete: try: if file.exists(): file.unlink() self.log(f"Deleted {file}") else: self.log(f"File {file} does not exist, skipping deletion") except Exception as e: self.log(f"Error deleting {file}: {str(e)}") if __name__ == "__main__": root = tk.Tk() app = JiraDataUtility(root) root.mainloop()
